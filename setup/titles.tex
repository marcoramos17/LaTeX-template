{1em}             % Spacing between number and title
{\_}


% --- Define a "chapter‐like" section style ---
% You can mimic the report/book chapter look by using the display shape,
% a big title, rules, and even a “Chapter” word.
\newcommand{\chapterSectionStyle}{%
	\titleformat{\section}[display]
	{\normalfont\huge}               % title format
	{%
		\vbox{%
			\vskip10ex% <-- extra space before the label
			\hbox to \linewidth{{Chapter \thesection}}%
		}%
	}
	{-30pt}                 % separation between label and title (set to 0pt)
	{%
		\vspace{1ex}\Huge\bfseries\MakeUppercase         % code before the title: rule, some space, centered
	}[\titlerule\vspace{6ex}]
	\preto\section{\clearpage}%
	
	
	\titleformat{\subsection}
	{\sectionfont\bfseries\Large} % Bold and large font for sections
	{\thesubsection.}    % Section number (optional)
	{1em}             % Spacing between number and title
	{\_}
	
	\titleformat{\subsubsection}
	{\sectionfont\bfseries\large} % Bold and large font for sections
	{\thesubsubsection.}    % Section number (optional)
	{1em}             % Spacing between number and title
	{\_}
}
